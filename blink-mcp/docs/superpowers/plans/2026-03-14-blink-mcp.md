**Goal:** Build a FastMCP stdio server that serves markdown prompt files from `~/.config/blink-mcp/prompts/` as MCP prompts, with filesystem watching for hot reload.

**Architecture:** Python server with entry point `server.py` and supporting modules (`prompt_loader.py`, `watcher.py`) split by responsibility. Reads `.md` files with YAML frontmatter, registers them as MCP prompts via FastMCP, and watches the directory for changes using `watchfiles`. A one-time migration script (`migrate.py`) seeds the prompts directory from existing opencode commands.

**Tech Stack:** Python 3.11+, FastMCP (mcp[cli]), watchfiles, python-frontmatter, uv for dependency management

```python
"""Load and parse prompt files from a directory."""
import logging
from pathlib import Path

import frontmatter

logger = logging.getLogger(__name__)


def load_prompt(filepath: Path) -> dict | None:
    """Load a single prompt file.

    Returns a dict with keys: name, description, body, has_arguments.
    Returns None if the file is malformed or missing required fields.
    """
    try:
        post = frontmatter.load(str(filepath))
    except Exception:
        logger.warning("Skipping malformed file: %s", filepath)
        return None

    description = post.metadata.get("description")
    if not description:
        logger.warning("Skipping file without description: %s", filepath)
        return None

    body = post.content
    has_arguments = "$ARGUMENTS" in body

    return {
        "name": filepath.stem,
        "description": description,
        "body": body,
        "has_arguments": has_arguments,
    }


def load_all_prompts(directory: Path) -> list[dict]:
    """Load all valid prompt files from a directory.

    Creates the directory if it doesn't exist.
    Returns a list of prompt dicts.
    """
    directory = Path(directory)
    if not directory.exists():
        logger.info("Creating prompts directory: %s", directory)
        directory.mkdir(parents=True, exist_ok=True)
        return []

    prompts = []
    for filepath in sorted(directory.glob("*.md")):
        prompt = load_prompt(filepath)
        if prompt is not None:
            prompts.append(prompt)

    return prompts
```

```python
"""blink-mcp: MCP prompt server for syncing slash commands across coding agents."""
import argparse
import logging
import os
from pathlib import Path

from mcp.server.fastmcp import FastMCP
from mcp.server.fastmcp.prompts import base
from mcp.types import GetPromptResult, PromptArgument, PromptMessage, TextContent

from prompt_loader import load_all_prompts

logger = logging.getLogger(__name__)

DEFAULT_PROMPTS_DIR = Path.home() / ".config" / "blink-mcp" / "prompts"


def create_server(prompts_dir: Path | None = None) -> FastMCP:
    """Create and configure the MCP server with prompts from disk."""
    if prompts_dir is None:
        prompts_dir = Path(
            os.environ.get("BLINK_MCP_PROMPTS_DIR", str(DEFAULT_PROMPTS_DIR))
        )
    prompts_dir = Path(prompts_dir)

    mcp = FastMCP("blink-mcp")

    prompts = load_all_prompts(prompts_dir)
    for prompt_data in prompts:
        _register_prompt(mcp, prompt_data)

    return mcp


def _register_prompt(mcp: FastMCP, prompt_data: dict) -> None:
    """Register a single prompt with the FastMCP server."""
    name = prompt_data["name"]
    description = prompt_data["description"]
    body = prompt_data["body"]
    has_arguments = prompt_data["has_arguments"]

    if has_arguments:
        @mcp.prompt(name=name, description=description)
        def prompt_fn(arguments: str, _body=body) -> str:
            return _body.replace("$ARGUMENTS", arguments)
    else:
        @mcp.prompt(name=name, description=description)
        def prompt_fn(_body=body) -> str:
            return _body


def main():
    """Entry point: parse args and run the server."""
    parser = argparse.ArgumentParser(description="blink-mcp prompt server")
    parser.add_argument(
        "--prompts-dir",
        type=Path,
        default=None,
        help="Path to prompts directory (default: ~/.config/blink-mcp/prompts/)",
    )
    args = parser.parse_args()

    server = create_server(args.prompts_dir)
    server.run(transport="stdio")


if __name__ == "__main__":
    main()
```

```python
"""Filesystem watcher for prompt directory changes."""
import asyncio
import logging
from collections.abc import Callable, Coroutine
from pathlib import Path
from typing import Any

from watchfiles import awatch, Change

logger = logging.getLogger(__name__)


class PromptWatcher:
    """Watches a directory for .md file changes and triggers a callback."""

    def __init__(
        self,
        directory: Path,
        on_change: Callable[[], Coroutine[Any, Any, None]],
    ):
        self.directory = directory
        self.on_change = on_change
        self._should_stop = False

    async def start(self) -> None:
        """Start watching for file changes."""
        logger.info("Watching for changes in: %s", self.directory)
        async for changes in awatch(self.directory, stop_event=self._stop_event):
            # Filter to only .md file changes
            md_changes = [
                (change_type, path)
                for change_type, path in changes
                if path.endswith(".md")
            ]
            if md_changes:
                logger.info("Detected prompt changes: %s", md_changes)
                await self.on_change()

    def stop(self) -> None:
        """Signal the watcher to stop."""
        self._stop_event.set()

    @property
    def _stop_event(self) -> asyncio.Event:
        if not hasattr(self, "_event"):
            self._event = asyncio.Event()
        return self._event
```

```python
# Add to server.py imports
import asyncio
from collections.abc import AsyncIterator
from contextlib import asynccontextmanager
from dataclasses import dataclass
from watcher import PromptWatcher


def create_server(prompts_dir: Path | None = None) -> FastMCP:
    """Create and configure the MCP server with prompts from disk."""
    if prompts_dir is None:
        prompts_dir = Path(
            os.environ.get("BLINK_MCP_PROMPTS_DIR", str(DEFAULT_PROMPTS_DIR))
        )
    prompts_dir = Path(prompts_dir)

    # Closure captures prompts_dir for the lifespan
    @asynccontextmanager
    async def app_lifespan(server: FastMCP) -> AsyncIterator[None]:
        """Start the filesystem watcher on server startup."""
        async def on_change():
            """Reload all prompts and notify clients."""
            logger.info("Reloading prompts from %s", prompts_dir)
            new_prompts = load_all_prompts(prompts_dir)
            # Clear existing prompts and re-register
            # NOTE: Verify FastMCP API during implementation — may need
            # server._prompt_manager._prompts.clear() or similar
            for prompt_data in new_prompts:
                _register_prompt(server, prompt_data)
            # Explicitly notify connected clients
            try:
                ctx = server.get_context()
                await ctx.session.send_prompt_list_changed()
            except Exception:
                logger.debug("No active session to notify")

        watcher = PromptWatcher(prompts_dir, on_change=on_change)
        task = asyncio.create_task(watcher.start())
        try:
            yield None
        finally:
            watcher.stop()
            task.cancel()
            try:
                await task
            except asyncio.CancelledError:
                pass

    mcp = FastMCP("blink-mcp", lifespan=app_lifespan)

    prompts = load_all_prompts(prompts_dir)
    for prompt_data in prompts:
        _register_prompt(mcp, prompt_data)

    return mcp
```

```python
"""One-time migration script: copy opencode commands to blink-mcp prompts.

Strips agent-specific frontmatter fields (agent, subtask) and keeps only
the description field and body content.

Usage:
    python migrate.py [--source ~/.config/opencode/command] [--dest ~/.config/blink-mcp/prompts]
"""
import argparse
import logging
from pathlib import Path

import frontmatter

logger = logging.getLogger(__name__)

STRIP_FIELDS = {"agent", "subtask"}

DEFAULT_SOURCE = Path.home() / ".config" / "opencode" / "command"
DEFAULT_DEST = Path.home() / ".config" / "blink-mcp" / "prompts"


def migrate_opencode_commands(source: Path, dest: Path) -> None:
    """Migrate all .md files from source to dest, stripping agent-specific fields."""
    dest.mkdir(parents=True, exist_ok=True)

    for filepath in sorted(source.glob("*.md")):
        try:
            post = frontmatter.load(str(filepath))
        except Exception:
            logger.warning("Skipping malformed file: %s", filepath)
            continue

        # Strip agent-specific fields
        for field in STRIP_FIELDS:
            post.metadata.pop(field, None)

        # Write cleaned file
        out_path = dest / filepath.name
        out_path.write_text(frontmatter.dumps(post))
        logger.info("Migrated: %s -> %s", filepath.name, out_path)


def main():
    logging.basicConfig(level=logging.INFO)
    parser = argparse.ArgumentParser(description="Migrate opencode commands to blink-mcp prompts")
    parser.add_argument("--source", type=Path, default=DEFAULT_SOURCE, help="Source command directory")
    parser.add_argument("--dest", type=Path, default=DEFAULT_DEST, help="Destination prompts directory")
    args = parser.parse_args()

    if not args.source.exists():
        logger.error("Source directory does not exist: %s", args.source)
        return

    migrate_opencode_commands(args.source, args.dest)
    logger.info("Migration complete. %d files migrated.", len(list(args.dest.glob("*.md"))))


if __name__ == "__main__":
    main()
```
